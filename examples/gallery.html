Quick Start

Get started in seconds

<!-- Include CSS -->
<link rel="stylesheet" href="src/css/swr.css">

<!-- Create roll container -->
<div id="myRoll"></div>

<!-- Include JS files -->
<script src="src/js/...all-files...></script>

<!-- Initialize -->
<script>
  const roll = new SWR('#myRoll', {
    aspectRatio: '9:16',
    autoplay: true,
    loop: true,
    items: [
      { type: 'image', src: 'image.jpg' },
      { type: 'video', src: 'video.mp4' },
      { type: 'html', content: '<div>...</div>' }
    ]
  });
</script>
